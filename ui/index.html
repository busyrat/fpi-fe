# 介绍

fpi-ui 是基于 element-ui 二次封装的UI库，可以直接在项目中使用 <el-xx />，再封装后的组件统一 <fpi-xx />

# 安装

# 安装依赖
yarn add git+ssh://git@git.fpi-inc.site:wanghao/fpi-ui.git

# 引入

// main.js
import Vue from 'vue';
import FpiUI from 'fpi-ui'
import App from './App.vue';

Vue.use(FpiUI);

new Vue({
  el: '#app',
  render: h => h(App)
});
